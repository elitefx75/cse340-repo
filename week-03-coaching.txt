// Global error handler  
app.use((err, req, res, next) => {  
    console.error('Error occurred:', err.message);  
    console.error('Stack trace:', err.stack);  

    const status = err.status || 500;  
    const template = status === 404 ? '404' : '500';  

    const context = {  
        title: status === 404 ? 'Page Not Found' : 'Server Error',  
        error: err.message,  
        stack: err.stack  
    };  

    res.status(status).render(`errors/${template}`, context);  
});  
For unexpected failures, the app still renders a 500 page instead of crashing. Behind the scenes, the server logs the error details so developers can debug without exposing internal stack traces to users.
